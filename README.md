### Consulta SQL

```sql
WITH alquileres_por_pelicula AS (
    SELECT
        c.name            AS categoria,
        f.film_id,
        f.title           AS titulo,
        COUNT(r.rental_id) AS total_alquileres
    FROM rental r
    INNER JOIN inventory  i  ON i.inventory_id  = r.inventory_id
    INNER JOIN film       f  ON f.film_id       = i.film_id
    INNER JOIN film_category fc ON fc.film_id   = f.film_id
    INNER JOIN category   c  ON c.category_id   = fc.category_id
    GROUP BY c.name, f.film_id, f.title
),
max_por_categoria AS (
    SELECT
        categoria,
        MAX(total_alquileres) AS max_alquileres
    FROM alquileres_por_pelicula
    GROUP BY categoria
)
SELECT
    ap.categoria,
    ap.titulo,
    ap.total_alquileres
FROM alquileres_por_pelicula ap
INNER JOIN max_por_categoria mc
    ON  mc.categoria     = ap.categoria
    AND mc.max_alquileres = ap.total_alquileres
ORDER BY ap.categoria ASC, ap.titulo ASC;
```

```sql
WITH gasto_por_cliente AS (
    SELECT
        c.customer_id,
        c.first_name,
        c.last_name,
        c.email,
        SUM(p.amount) AS gasto_total
    FROM customer c
    INNER JOIN payment p ON p.customer_id = c.customer_id
    GROUP BY c.customer_id, c.first_name, c.last_name, c.email
)
SELECT
    first_name AS nombre,
    last_name AS apellido,
    email AS correo,
    gasto_total,
    (SELECT AVG(gasto_total) FROM gasto_por_cliente) AS promedio
FROM gasto_por_cliente
WHERE gasto_total > (SELECT AVG(gasto_total) FROM gasto_por_cliente)
ORDER BY gasto_total DESC;
```

### Consulta SQL
```sql
WITH alquileres_por_pelicula AS (
    SELECT
        fc.category_id,
        c.name            AS categoria,
        f.film_id,
        f.title,
        COUNT(r.rental_id) AS total_alquileres
    FROM rental r
    INNER JOIN inventory      i  ON i.inventory_id  = r.inventory_id
    INNER JOIN film           f  ON f.film_id       = i.film_id
    INNER JOIN film_category  fc ON fc.film_id      = f.film_id
    INNER JOIN category       c  ON c.category_id   = fc.category_id
    GROUP BY fc.category_id, c.name, f.film_id, f.title
),
promedio_por_categoria AS (
    SELECT
        category_id,
        AVG(total_alquileres) AS promedio_alquileres
    FROM alquileres_por_pelicula
    GROUP BY category_id
)
SELECT
    ap.categoria,
    ap.title AS titulo,
    ap.total_alquileres,
    pc.promedio_alquileres AS promedio_categoria
FROM alquileres_por_pelicula ap
INNER JOIN promedio_por_categoria pc
    ON pc.category_id = ap.category_id
WHERE ap.total_alquileres > pc.promedio_alquileres
ORDER BY ap.categoria ASC, ap.total_alquileres DESC;
```
